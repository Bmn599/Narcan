    <section id="recognize" class="wrap">

      <h2 class="section-title">Recognize the signs of an opioid overdose</h2>

      <p class="sub">If you suspect an overdose, call 911. Then follow the steps below.</p>

      <div class="cards">

        <article class="card span-6">

          <h3>Common signs</h3>

          <ul>

            <li>Unresponsive, cannot be woken up</li>

            <li>Slow, shallow, or no breathing, gurgling or snoring sounds</li>

            <li>Pale, clammy skin, blue or gray lips or fingertips</li>

            <li>Pinpoint pupils</li>

          </ul>

        </article>

        <article class="card span-6">

          <h3>What to do first</h3>

          <ul>

            <li>Call <strong>911</strong> and state possible opioid overdose</li>

            <li>Lay the person on their back and clear the airway</li>

            <li>Be ready to start rescue breathing if needed</li>

            <li>Give naloxone as soon as it is available</li>

          </ul>

        </article>

      </div>

    </section>


    <section id="use" class="wrap">

      <div class="grid-2">

        <div>

          <h2 class="section-title">How to use Narcan nasal spray 4 mg</h2>

          <p class="sub">One device equals one dose. If there is no response in 2 to 3 minutes, give another dose with a new device.</p>

          <div class="steps">

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Call 911</strong>. If available, ask someone nearby to grab naloxone and an AED.</div></div>

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Lay on back, tilt head</strong>. Support the neck. Insert tip into a nostril until your fingers touch the nose.</div></div>

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Press plunger firmly</strong> to release the dose. Remove the device after spraying.</div></div>

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Start rescue breathing</strong> if not breathing normally. Give 1 breath every 5 to 6 seconds. If trained and needed, provide chest compressions.</div></div>

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Repeat in 2 to 3 minutes</strong> with a new device if there is little or no response or if breathing slows again.</div></div>

            <div class="step"><div class="badge" aria-hidden="true"></div><div><strong>Stay with the person</strong> until help arrives. Place them in the recovery position if they start breathing on their own.</div></div>

          </div>

          <p class="note" style="margin-top:10px">Naloxone has a short effect. Symptoms can return. Monitor closely and be ready to give another dose.</p>

        </div>

        <aside class="card" aria-label="Important safety notes">

          <span class="pill" aria-hidden="true">Important</span>

          <ul>

            <li>Naloxone will not harm someone if opioids are not present.</li>

            <li>Store at room temperature. Check expiration dates.</li>

            <li>Good Samaritan protections vary by state. Know your local policy.</li>

          </ul>

        </aside>

      </div>

    <section id="faq" class="wrap">

      <h2 class="section-title">Frequently asked questions</h2>

      <details>

        <summary>Can naloxone be used on someone who is not overdosing on opioids</summary>

        <p>Yes. Naloxone targets opioid receptors and will not harm someone if opioids are not involved.</p>

      </details>

      <details>

        <summary>How long does naloxone last</summary>

        <p>Its effects can wear off in 30 to 90 minutes. Breathing may slow again. Monitor and repeat doses as needed.</p>

      </details>

      <details>

        <summary>What should I expect after giving naloxone</summary>

        <p>The person may wake up disoriented or in withdrawal. Keep them safe, explain what happened, and wait for EMS.</p>

      </details>

    </section>

      const walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, {

        acceptNode: (node) => {

          // Skip whitespace-only nodes and nodes inside scripts/styles

          if(!node.nodeValue.trim()) return NodeFilter.FILTER_REJECT;

          let p = node.parentNode;

          while(p){

          }

          return NodeFilter.FILTER_ACCEPT;

        }

      });

      const toUpdate = [];

      let n;

      while(n = walker.nextNode()){ toUpdate.push(n); }

      toUpdate.forEach(node=>{

        let t = node.nodeValue;

        // Replace space dash space, space en dash space, space em dash space with single space

        t = t.replace(/\s[-–—]\s/g, ' ');

        // Replace standalone em/en dashes with a space

        t = t.replace(/[–—]/g, ' ');

        node.nodeValue = t;

      });

    })();
